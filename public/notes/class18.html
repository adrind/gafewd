Build a website

• What if we wanted to add new data?

$.ajax({
	url: 'https://api.airtable.com/v0/appQytW1PVY1M77vN/Favorites',
	headers: {
		'Authorization': 'Bearer keyHSu1cfEZyaBVMg'
	},
	type: 'POST',
	data: {
		fields: {
			text: 'Chocolate',
			name: 'Adrienne',
			image: 'picture-of-chocolate.jpg'
		}
	},
	success: function(response) {
		alert('Data added!');
	}
})
